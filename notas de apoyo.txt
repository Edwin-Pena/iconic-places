//aparicion de notas  cada vez que me paro encima de ellas
const notas = document.querySelectorAll(".note__container");

notas.forEach((note, index) => {
  const image = note.querySelector(".gallery__image");
  const noteData = notes[index];
  const title = document.createElement("h3");
  const description = document.createElement("p");

  title.classList.add("note__title");
  description.classList.add("note__desc");

  title.innerText = `${noteData.name}`;
  description.innerText = `${noteData.description}`;

  const selectElement = note.querySelector(".note");
  selectElement.appendChild(title);
  selectElement.appendChild(description);

  image.addEventListener("mouseover", () => {
    selectElement.style.display = "flex";
  });

  image.addEventListener("mouseout", () => {
    selectElement.style.display = "none";
  });
});


//codigo del navbar para que se vea el continente seleccionado
const seleccionarContinentes = document.querySelectorAll(".menu__options");
const all = document.querySelector(".all");
const asia = document.querySelector(".asia");
const europe = document.querySelector(".europe");
const northAmerica = document.querySelector(".north-america");
const southAmerica = document.querySelector(".south-america");

seleccionarContinentes.forEach((opcion) => {
  opcion.addEventListener("click", (e) => {
    e.preventDefault();

    // Remove active class from all options
    seleccionarContinentes.forEach((opcion) => {
      opcion.classList.remove("menu__options--active");
    });

    // Add active class to the clicked option
    opcion.classList.add("menu__options--active");

    /* la validacion de cual seccion se debe mostrar debe hacerse desde el evento de clic para que funcione ya que yo lo estaba haciendo por fuera y este no iba a servir */
    if (opcion.innerText === "All") { //En este caso se puede validar que un elemento tenga un texto en especifico con innerText
      all.style.display = "grid";
      asia.style.display = "none";
      europe.style.display = "none";
      northAmerica.style.display = "none";
      southAmerica.style.display = "none";
    } else if (opcion.innerText === "Asia") {
      all.style.display = "none";
      asia.style.display = "flex";
      europe.style.display = "none";
      northAmerica.style.display = "none";
      southAmerica.style.display = "none";
    } else if (opcion.innerText === "Europe") {
      all.style.display = "none";
      asia.style.display = "none";
      europe.style.display = "flex";
      northAmerica.style.display = "none";
      southAmerica.style.display = "none";
    } else if (opcion.innerText === "North America") {
      all.style.display = "none";
      asia.style.display = "none";
      europe.style.display = "none";
      northAmerica.style.display = "flex";
      southAmerica.style.display = "none";
    } else if (opcion.innerText === "South America") {
      all.style.display = "none";
      asia.style.display = "none";
      europe.style.display = "none";
      northAmerica.style.display = "none";
      southAmerica.style.display = "flex";
    }
  });
});
